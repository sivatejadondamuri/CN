Aim:
To write a C program to implement the Go-Back-N sliding window protocol for reliable data transmission in the data link layer.

Theory:
Go-Back-N is a sliding window protocol where the sender can send multiple frames (up to window size) without waiting for acknowledgements. If a frame is lost or an error occurs, the receiver discards all subsequent frames and sends a negative acknowledgement. The sender then retransmits the erroneous frame and all frames after it. This ensures reliable, ordered delivery of frames.

Program (C):

#include <stdio.h>

int main() {
    int window_size, total_frames, i, j, error;

    printf("Enter total number of frames: ");
    scanf("%d", &total_frames);

    printf("Enter window size: ");
    scanf("%d", &window_size);

    i = 1;
    while (i <= total_frames) {

        printf("\nSender sends frames: ");
        for (j = i; j < i + window_size && j <= total_frames; j++)
            printf("%d ", j);

        printf("\nEnter error frame number (0 for no error): ");
        scanf("%d", &error);

        if (error >= i && error < i + window_size) {
            printf("Error detected at frame %d\n", error);
            printf("Receiver rejects from frame %d\n", error);
            printf("Sender retransmits from frame %d\n", error);
            i = error;  
        } else {
            printf("All frames acknowledged from %d to %d\n", i, j - 1);
            i = j;  
        }
    }

    printf("\nAll frames sent successfully.\n");

    return 0;
}
